    function isOrthonormal4x4(M) {
      let orthogonal = true;
      for (let i = 0; i < 4; i++) {
        for (let j = i+1; j < 4; j++) {
          let sum = 0.0;
          for (let k = 0; k < 4; k++) {
            sum += M[i*4+k]*M[j*4+k];
          }
          if (Math.abs(sum) > 1e-5) {
            orthogonal = false;
          }
        }
      }
      let normal = true;
      for (let i = 0; i < 4; i++) {
        let sum = 0.0;
        for (let j = 0; j < 4; j++) {
          sum += M[i*4+j]*M[i*4+j];
        }
        if (Math.abs(sum-1) > 1e-3) {
          normal = false;
        }
      }
      return orthogonal && normal;
    }
